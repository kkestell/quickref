```python
import math

# Constants
pi = math.pi
e = math.e

# Basic math functions
square_root = math.sqrt(4)
absolute_value = math.fabs(-4)
floor = math.floor(3.7)
ceiling = math.ceil(3.2)
power = math.pow(2, 3)
exponential = math.exp(2)
logarithm = math.log(10)
log_base_2 = math.log2(8)

# Trigonometric functions
sine = math.sin(math.pi/2)
cosine = math.cos(math.pi/2)
tangent = math.tan(math.pi/4)
inverse_sine = math.asin(1)
inverse_cosine = math.acos(0)
inverse_tangent = math.atan(1)

# Hyperbolic functions
hyperbolic_sine = math.sinh(0)
hyperbolic_cosine = math.cosh(0)
hyperbolic_tangent = math.tanh(0)

# Angular conversion
degrees_to_radians = math.radians(180)
radians_to_degrees = math.degrees(math.pi)

# Statistical functions
mean = math.mean([1, 2, 3])
median = math.median([1, 2, 3])
mode = math.mode([1, 2, 2, 3])

# Random functions
random_number = math.random()
random_integer = math.randint(1, 10)

# Distance and angle functions
distance = math.dist([0, 0], [3, 4])
```
